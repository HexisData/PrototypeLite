Procedure Source Code

CREATE FUNCTION [hq_Core].[tvf_Split_DbObjName]
(
   @String NVARCHAR(MAX)
)
RETURNS TABLE

RETURN
(
    SELECT
        COALESCE(C.Item, D.Item) AS Item
    FROM
        hq_Core.tvf_Split_Delimiter(@String, '_') D
        OUTER APPLY (
            SELECT Item 
            FROM hq_Core.tvf_Split_CamelCase(D.Item)
            WHERE D.Item <> UPPER(D.Item) COLLATE Latin1_General_CS_AS
        ) C
)
